1//! Traits and common behaviors for raytracer objects
2//! 
3//! This module defines the core traits that all renderable objects in the raytracer
4//! must implement, as well as an enum for polymorphic handling of different object types.
5//! It serves as the foundation for the object behavior system in the raytracer.
6//! 
7//! author:         Zhean Ganituen (zrygan)
8//! last updated:   April 16, 2025
9
10use super::circle::ObjectCircle;
11use super::emitters::{Emitter, EmitterCollimated};
12
13/// Enum that represents all possible object types in the raytracer.
14///
15/// This enum allows for polymorphic handling of different object types
16/// in collections and the rendering system. Each variant wraps a concrete
17/// object implementation.
18pub enum RaytracerObjects {
19    /// A circular shape object (used for barriers, mirrors, etc.)
20    ObjectCircle(ObjectCircle),
21    /// A standard isotropic light emitter
22    Emitter(Emitter),
23    /// A specialized light emitter that produces parallel rays
24    EmitterCollimated(EmitterCollimated),
25}
26
27/// Trait for objects that can be rendered to the screen.
28///
29/// All visible objects in the raytracer must implement this trait,
30/// which provides the necessary method to render the object.
31pub trait Drawable {
32    /// Renders the object to the screen.
33    ///
34    /// Each implementing type will define its own rendering behavior
35    /// appropriate to its visual representation.
36    fn draw_object(&self);
37}
38
39/// Trait for objects that can be moved to a new position.
40///
41/// Provides functionality for interactive placement and animation
42/// of objects within the raytracer environment.
43pub trait Movable {
44    /// Moves the object to a new position.
45    ///
46    /// # Arguments
47    ///
48    /// * `pos_x` - The new x-coordinate position
49    /// * `pos_y` - The new y-coordinate position 
50    fn move_object(&mut self, pos_x: f32, pos_y: f32);
51}
