10/// Converts a vector (angle, start point) to a pair of points.
11///
12/// This function takes a vector defined by an angle and magnitude, and calculates
13/// the end coordinates when that vector is applied from a given start point.
14/// This is useful for ray calculations and object placement.
15///
16/// # Arguments
17///
18/// * `theta` - The angle in degrees (Macroquad uses degrees for angles)
19/// * `magnitude` - The magnitude of the vector, typically (screen_width, screen_height)
20/// * `start_point` - The initial point (x, y) from which the vector extends
21///
22/// # Returns
23///
24/// A tuple containing four values:
25/// * The x-coordinate of the start point
26/// * The y-coordinate of the start point
27/// * The x-coordinate of the end point 
28/// * The y-coordinate of the end point
29///
30/// # Example
31///
32/// ```
33/// // Calculate end point for a ray starting at (400, 300) with 45 degree angle
34/// let (start_x, start_y, end_x, end_y) = vec_to_coords(
35///     45.0,
36///     (screen_width(), screen_height()),
37///     (400.0, 300.0)
38/// );
39/// ```
40pub fn vec_to_coords(theta: f32, magnitude: (f32, f32), start_point: (f32, f32)) -> (f32, f32, f32, f32) {
41    (
42        start_point.0,
43        start_point.1,
44        start_point.0 + magnitude.0 * theta.cos(),
45        start_point.1 + magnitude.0 * theta.sin(),
46    )
47}
