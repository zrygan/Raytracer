1//! Emitter objects initialization and behaviors
2//!
3//! This module provides light emitter implementations for the raytracer system.
4//! It defines two types of emitters: isotropic (radiating in all directions)
5//! and collimated (parallel rays, like a laser).

17/// Enumeration of all emitter types supported by the raytracer.
18///
19/// This enum allows different emitter types to be treated polymorphically
20/// in the rendering and physics systems.
21pub enum Emitters {
22    /// Standard isotropic emitter that radiates light in all directions
23    Emitter(Emitter),
24    /// Directional emitter that produces parallel light rays
25    EmitterCollimated(EmitterCollimated),
26}
27
28impl Drawable for Emitters {
29    /// Draws the emitter on the screen.
30    ///
31    /// Delegates to the appropriate draw_object method based on the emitter type.
32    fn draw_object(&self) {
33        match self {
34            Emitters::Emitter(e) => e.draw_object(),
35            Emitters::EmitterCollimated(e) => e.base_emitter.draw_object(),
36        }
37    }
38}
39
40impl Movable for Emitters {
41    /// Moves the emitter to a new position.
42    ///
43    /// Delegates to the appropriate move_object method based on the emitter type.
44    ///
45    /// # Arguments
46    ///
47    /// * `pos_x` - The new x-coordinate position
48    /// * `pos_y` - The new y-coordinate position
49    // #[warn(unused_variables)]
50    fn move_object(&mut self, pos_x: f32, pos_y: f32) {
51        match self {
52            Emitters::Emitter(e) => e.move_object(pos_x, pos_y),
53            Emitters::EmitterCollimated(e) => e.base_emitter.move_object(pos_x, pos_y),
54        }
55    }
56}

58impl Drawable for Emitter {
59    /// Draws the isotropic emitter (or the base emitter) and its rays on the 
60    /// screen.
61    ///
62    /// Renders the emitter as a colored circle and draws all of its
63    /// associated light rays emanating from it.
64    fn draw_object(&self) {
65        // Draw the emitter's physical representation (a circle)
66        draw_circle(
67            self.base_object.pos_x,
68            self.base_object.pos_y,
69            self.base_object.radius,
70            self.base_object.color_fill,
71        );
72
73        // Draw all the light rays associated with this emitter
74        for ray in self.rays.clone() {
75            draw_line(
76                ray.start_x,
77                ray.start_y,
78                ray.end_x,
79                ray.end_y,
80                ray.thickness,
81                ray.color,
82            );
83        }
84    }
85}

103/// Represents a standard isotropic light emitter.
104///
105/// This emitter radiates light rays in all directions from a central point,
106/// similar to a point light source in real-world physics.
107#[derive(Clone)]
108pub struct Emitter {
109    /// The physical representation of the emitter (position, size, color)
110    pub base_object: ObjectCircle,
111    /// Collection of light rays emanating from this emitter
112    pub rays: Vec<ObjectRay>,
113}
114
115impl Emitter {
116    /// Creates a new isotropic emitter with the given properties.
117    ///
118    /// Adds the newly created emitter to the global object collection
119    /// for tracking in the raytracer system.
120    ///
121    /// # Arguments
122    ///
123    /// * `base_object` - The physical properties of the emitter
124    /// * `rays` - Collection of rays to be emitted from this source
125    ///
126    /// # Returns
127    ///
128    /// A new `Emitter` instance with the specified parameters
129    pub fn new(base_object: ObjectCircle, rays: Vec<ObjectRay>) -> Emitter {
130        // if rays.len() > OBJC_MAX_RAY_COUNT as usize {
131        //     panic!("Raytracer: Emitter has too many rays. See OBJC_MAX_RAY_COUNT in globals.")
132        // }
133
134        let new_emitter: Emitter = Emitter { base_object, rays };
135
136        OBJ_COLLECTION
137            .lock()
138            .unwrap()
139            .push(RaytracerObjects::Emitter(new_emitter.clone()));
140
141        new_emitter
142    }
143}

145/// Represents a collimated (parallel rays) light emitter.
146///
147/// This emitter produces rays that travel in parallel, similar to a laser beam
148/// in real-world physics. It has an orientation and beam diameter.
149#[derive(Clone)]
150pub struct EmitterCollimated {
151    /// The underlying emitter providing basic functionality
152    pub base_emitter: Emitter,
153    /// The angle (in radians) at which rays are emitted
154    pub orientation: f32,
155    /// The width of the beam of parallel rays
156    pub collimated_beam_diameter: f32,
157}
158
159impl EmitterCollimated {
160    /// Creates a new collimated emitter with the specified properties.
161    ///
162    /// Adds the newly created emitter to the global object collection
163    /// for tracking in the raytracer system.
164    ///
165    /// # Arguments
166    ///
167    /// * `base_object` - The physical properties of the emitter
168    /// * `rays` - Collection of rays to be emitted from this source
169    /// * `orientation` - The angle (in radians) at which rays are emitted
170    /// * `collimated_beam_diameter` - The width of the beam of parallel rays
171    ///
172    /// # Returns
173    ///
174    /// A new `EmitterCollimated` instance with the specified parameters
175    pub fn new(
176        base_object: ObjectCircle,
177        rays: Vec<ObjectRay>,
178        orientation: f32,
179        collimated_beam_diameter: f32,
180    ) -> EmitterCollimated {
181        let new_emitter = EmitterCollimated {
182            base_emitter: Emitter { base_object, rays },
183            orientation,
184            collimated_beam_diameter,
185        };
186
187        OBJ_COLLECTION
188            .lock()
189            .unwrap()
190            .push(RaytracerObjects::EmitterCollimated(new_emitter.clone()));
191
192        new_emitter
193    }
