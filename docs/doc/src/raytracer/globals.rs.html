1//! Contains all global variables for the Raytracer project
2//!
3//! This module centralizes configuration constants and shared resources for the raytracer.
4//! It includes application metadata, window settings, visual defaults, keybindings,
5//! and object limitations used throughout the application.
6
7use crate::objects::behavior::RaytracerObjects;
8use macroquad::input::KeyCode::{self};
9use macroquad::prelude::Color;
10use once_cell::sync::Lazy;
11use std::sync::Mutex;
12
13/// App Information (starts with the APP_ prefix)
14/// 
15/// These constants contain metadata about the application itself that may be
16/// displayed in window titles, about screens, or for identifying the application.
17pub const APP_NAME: &str = "Raytracer";
18pub const APP_VERSION: &str = "v0.rp";
19pub const APP_AUTHOR: &str = "Zhean Ganituen";
20pub const APP_GITHUB: &str = "https://github.com/zrygan/raytracer";
21// pub const APP_ICON: Icon;
22
23/// Window Settings (starts with the WINDOW_ prefix)
24/// 
25/// These constants define the visual appearance and dimensions of the application window.
26pub const WINDOW_HEIGHT: i32 = 800;
27pub const WINDOW_WIDTH: i32 = 600;
28pub const WINDOW_BG_COLOR: Color = Color::new(0.00, 0.00, 0.00, 1.00); // Black
29
30/// Standard Colors
31/// 
32/// Common colors used throughout the application for consistent visual styling.
33pub const CORNFLOWER_BLUE: Color = Color::new(1.00, 0.34, 0.20, 1.00);
34
35/// Macroquad Set Up (starts with the MACROQUAD_ prefix)
36/// 
37/// These constants configure the Macroquad rendering engine behavior including
38/// display quality options and window properties.
39pub const MACROQUAD_HIGH_DPI: bool = false;
40pub const MACROQUAD_FULLSCREEN: bool = false;
41pub const MACROQUAD_SAMPLE_COUNT: i32 = 0; // MSAA sample count
42pub const MACROQUAD_RESIZEABLE: bool = false;
43
44/// Raytracer Object Collection
45/// 
46/// Thread-safe global collection of all objects in the raytracer scene.
47/// Uses a mutex to allow safe mutation from different parts of the code.
48pub static OBJ_COLLECTION: Lazy<Mutex<Vec<RaytracerObjects>>> =
49    Lazy::new(|| Mutex::new(Vec::new()));
50
51/// Raytracer Object Constants (starts with the OBJC_ prefix)
52/// 
53/// These constants define limitations for object counts to prevent performance issues
54/// and memory overflow.
55pub const OBJC_MAX_OBJ_COUNT: i32 = 100;
56pub const OBJC_MAX_RAY_COUNT: i32 = 25;
57
58/// Raytracer Default Object Parameters (starts with OBJD_ prefix)
59/// 
60/// These constants define the default visual appearance and dimensions of
61/// raytracer objects when created.
62pub const OBJD_CIRCLE_RADIUS: f32 = 50.0;
63pub const OBJD_CIRCLE_FILL: Color = CORNFLOWER_BLUE;
64pub const OBJD_RAY_WIDTH: f32 = 2.5;
65pub const OBJD_RAY_COLOR: Color = CORNFLOWER_BLUE;
66
67/// Raytracer Keybinds (starts with KEYB_ prefix)
68/// 
69/// These constants map keyboard keys to specific actions in the raytracer,
70/// making it easy to modify keybindings from a central location.
71pub const KEYB_SIMPLE_CIRCLE: KeyCode = KeyCode::S;
72pub const KEYB_EMITTER_ISOTROPIC: KeyCode = KeyCode::I;
73pub const KEYB_EMITTER_COLLIMATED: KeyCode = KeyCode::C;
